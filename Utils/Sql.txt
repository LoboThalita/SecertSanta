CREATE TABLE Game (
    Id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    Name VARCHAR2(50)
);
CREATE TABLE Player (
    Id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    Name VARCHAR2(50) NOT NULL,
    Email VARCHAR2(100),
    Whatsapp VARCHAR2(20),
    GameId NUMBER NOT NULL,
    FriendId NUMBER,
    CONSTRAINT fk_Game FOREIGN KEY (GameId) REFERENCES Game(Id),
    CONSTRAINT fk_Friend FOREIGN KEY (FriendId) REFERENCES Player(Id)
);

CREATE TABLE Client (
    Id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    Email VARCHAR2(100) NOT NULL UNIQUE,
    Password VARCHAR2(100) NOT NULL,
    Name VARCHAR2(50) NOT NULL
);
